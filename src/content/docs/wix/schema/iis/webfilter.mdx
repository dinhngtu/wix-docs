IIs Filter for a Component or Web Site


## Attributes

<BlockWrapper>

  <BlockReference
    content={{
      title: "Description",
      type: "string"
    }}
  >
  Description of the filter.
  </BlockReference>

  <BlockReference
    content={{
      title: "Flags",
      type: "wxs:Integer"
    }}
  >
  Sets the MD_FILTER_FLAGS metabase key for the filter. This must be an integer. See MSDN 'FilterFlags' documentation for more details.
  </BlockReference>

  <BlockReference
    content={{
      title: "Id",
      type: "string"
    }}
  >
  The unique Id for the web filter. If the Id is not specified, it will be generated.
  </BlockReference>

  <BlockReference
    content={{
      title: "LoadOrder",
      type: "string"
    }}
  >
  The legal values are "first", "last", or a number.
   If a number is specified, it must be greater than 0.
  </BlockReference>

  <BlockReference
    content={{
      title: "Name",
      type: "string",
      required: true
    }}
  >
  The name of the filter to be used in IIS.
  </BlockReference>

  <BlockReference
    content={{
      title: "Path",
      type: "string",
      required: true
    }}
  >
  The path of the filter executable file.
   This should usually be a value like '[!FileId]', where 'FileId' is the file identifier
   of the filter executable file.
  </BlockReference>

  <BlockReference
    content={{
      title: "WebSite",
      type: "string"
    }}
  >
  Specifies the parent website for this filter (if there is one).
   If this is a global filter, then this attribute should not be specified.
  </BlockReference>

</BlockWrapper>
