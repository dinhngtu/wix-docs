Searches for an installation of .NET Core SDK.
 The variable gets set to the latest version of the given Platform and Major Version.


## Attributes

<BlockWrapper>

  <BlockReference
    content={{
      title: "After",
      type: "string"
    }}
  >
  Id of the search that this one should come after.
  </BlockReference>

  <BlockReference
    content={{
      title: "Condition",
      type: "string"
    }}
  >
  Condition for evaluating the search. If this evaluates to false, the search is not executed at all.
  </BlockReference>

  <BlockReference
    content={{
      title: "Id",
      type: "string"
    }}
  >
  Identifier for the search. If the Id is not specified, one will be generated.
  </BlockReference>

  <BlockReference
    content={{
      title: "MajorVersion",
      type: "wxs:Integer"
    }}
  >
  Major version of .NET SDK. For example, "6".
  </BlockReference>

  <BlockReference
    content={{
      title: "Platform",
      type: "NetfxPreprocessorPlatformType",
      typeLink: "../netfxpreprocessorplatformtype/",
      typeTitle: "Platform referencing the machine architecture."
    }}
  >
  The platform to search for.
  </BlockReference>

  <BlockReference
    content={{
      title: "Variable",
      type: "string",
      required: true
    }}
  >
  Name of the variable in which to place the result of the search.
  </BlockReference>

</BlockWrapper>
