Extension for a Component


## Attributes

<BlockWrapper>

  <BlockReference
    content={{
      title: "Advertise",
      type: "yesno"
    }}
  >
  Whether this extension is to be advertised. The default is "no".
  </BlockReference>

  <BlockReference
    content={{
      title: "ContentType",
      type: "string"
    }}
  >
  The MIME type that is to be written.
  </BlockReference>

  <BlockReference
    content={{
      title: "Id",
      type: "string",
      required: true
    }}
  >
  This is simply the file extension, like "doc" or "xml". Do not include the preceding period.
  </BlockReference>

</BlockWrapper>
