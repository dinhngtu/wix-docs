import BlockWrapper from '@components/BlockWrapper.astro';
import BlockReference from '@components/BlockReference.astro';
import BlockDefList from '@components/BlockDefList.astro';

Describes a single msp package to install.


## Attributes

<BlockWrapper>

  <BlockReference
    content={{
      title: "After",
      type: "string"
    }}
  >
  The identifier of another package that this one should be installed after. By default the After
   attribute is set to the previous sibling package in the Chain or PackageGroup element. If this
   attribute is specified ensure that a cycle is not created explicitly or implicitly.
  </BlockReference>

  <BlockReference
    content={{
      title: "Cache",
      type: "KeepRemoveForceTypeUnion",
      typeLink: "../keepremoveforcetype",
      typeTitle: "Values of this type will either be \"force\", \"keep\", or \"remove\"."
    }}
  >
  Whether to cache the package. The default is "keep".
  </BlockReference>

  <BlockReference
    content={{
      title: "CacheId",
      type: "string"
    }}
  >
  The identifier to use when caching the package.
  </BlockReference>

  <BlockReference
    content={{
      title: "Compressed",
      type: "YesNoDefaultTypeUnion",
      typeLink: "../yesnodefaulttype",
      typeTitle: "Values of this type will either be \"default\", \"yes\", or \"no\"."
    }}
  >
  Whether the package payload should be embedded in a container or left as an external payload.
   Must not be specified if the package payload is given by the *PackagePayload element.
  </BlockReference>

  <BlockReference
    content={{
      title: "Description",
      type: "string"
    }}
  >
  Specifies the description to place in the bootstrapper application data manifest for the package. By default, ExePackages
   use the FileName field from the version information, MsiPackages use the ARPCOMMENTS property, and MspPackages use
   the Description patch metadata property. Other package types must use this attribute to define a description in the
   bootstrapper application data manifest.
  </BlockReference>

  <BlockReference
    content={{
      title: "DisplayInternalUICondition, from extension namespace http://wixtoolset.org/schemas/v4/wxs/bal",
      type: "string"
    }}
  >
  Specifies whether the bundle will show the UI authored into the msi package. If not specified or the condition evaluates
   to false, all information is routed to the bootstrapper application to provide a unified installation experience.
   Otherwise, the UI authored into the msi package will be displayed on top of any bootstrapper application UI.
   WixStdBA doesn't support EmbeddedUI.
  </BlockReference>

  <BlockReference
    content={{
      title: "DisplayName",
      type: "string"
    }}
  >
  Specifies the display name to place in the bootstrapper application data manifest for the package. By default, ExePackages
   use the ProductName field from the version information, MsiPackages use the ProductName property, and MspPackages use
   the DisplayName patch metadata property. Other package types must use this attribute to define a display name in the
   bootstrapper application data manifest.
  </BlockReference>

  <BlockReference
    content={{
      title: "DownloadUrl",
      type: "string"
    }}
  >
  The URL to use to download the package. The following substitutions are supported:
  
  
  - \{0} is replaced by the package Id.
  - \{1} is replaced by the payload Id.
  - \{2} is replaced by the payload file name.
  Must not be specified if the package payload is given by the *PackagePayload element.
  </BlockReference>

  <BlockReference
    content={{
      title: "Id",
      type: "string"
    }}
  >
  Identifier for this package, for ordering and cross-referencing. The default is the Name attribute
   modified to be suitable as an identifier (i.e. invalid characters are replaced with underscores).
  </BlockReference>

  <BlockReference
    content={{
      title: "InstallCondition",
      type: "string"
    }}
  >
  A condition to evaluate before installing the package. The package will only be installed if the condition
   evaluates to true. If the condition evaluates to false and the bundle is being installed, repaired, or modified,
   the package will be uninstalled.
  </BlockReference>

  <BlockReference
    content={{
      title: "InstallSize",
      type: "string"
    }}
  >
  The size this package will take on disk in bytes after it is installed. By default, the binder will
   calculate the install size by scanning the package (File table for MSIs, Payloads for EXEs)
   and use the total for the install size of the package.
  </BlockReference>

  <BlockReference
    content={{
      title: "LogPathVariable",
      type: "string"
    }}
  >
  Name of a Variable that will hold the path to the log file. An empty value will cause the variable to not
   be set. The default is "WixBundleLog_[PackageId]" except for MSU packages which default to no logging.
  </BlockReference>

  <BlockReference
    content={{
      title: "Name",
      type: "string"
    }}
  >
  The destination path and file name for this chain payload. Use this attribute to rename the
   chain entry point or extract it into a subfolder. The default value is the file name from the
   SourceFile attribute, if provided. At a minimum, the Name or SourceFile attribute must be specified.
   This must be a relative path ('\foo' or 'C:\foo' is not allowed).
   Must not be specified if the package payload is given by the *PackagePayload element.
  </BlockReference>

  <BlockReference
    content={{
      title: "PerMachine",
      type: "YesNoDefaultTypeUnion",
      typeLink: "../yesnodefaulttype",
      typeTitle: "Values of this type will either be \"default\", \"yes\", or \"no\"."
    }}
  >
  Indicates the package must be executed elevated. The default is "no".
  </BlockReference>

  <BlockReference
    content={{
      title: "Permanent",
      type: "yesno"
    }}
  >
  Specifies whether the package can be uninstalled. The default is "no".
  </BlockReference>

  <BlockReference
    content={{
      title: "PrereqLicenseFile, from extension namespace http://wixtoolset.org/schemas/v4/wxs/bal",
      type: "string"
    }}
  >
  Source file of the RTF license file.
   There may only be one package in the bundle that has either the PrereqLicenseFile attribute or the PrereqLicenseUrl attribute.
  </BlockReference>

  <BlockReference
    content={{
      title: "PrereqLicenseUrl, from extension namespace http://wixtoolset.org/schemas/v4/wxs/bal",
      type: "string"
    }}
  >
  URL target of the license link.
   There may only be one package in the bundle that has either the PrereqLicenseFile attribute or the PrereqLicenseUrl attribute.
  </BlockReference>

  <BlockReference
    content={{
      title: "PrereqPackage, from extension namespace http://wixtoolset.org/schemas/v4/wxs/bal",
      type: "wxs:YesNoTypeUnion"
    }}
  >
  When set to "yes", the Prereq BA will plan the package to be installed if its InstallCondition is "true" or empty.
  </BlockReference>

  <BlockReference
    content={{
      title: "PrimaryPackageType, from extension namespace http://wixtoolset.org/schemas/v4/wxs/bal",
      type: ""
    }}
  >
  Used by WixInternalUIBootstrapperApplication to determine which package is the primary package. The default value is "default". This attribute's value must be one of the following:
  - *default*: This package is used unless there is a package for the runtime machine's architecture.
  - *x86*: This package is used when the runtime machine's OS is x86 (IMAGE_FILE_MACHINE_I386).
  - *x64*: This package is used when the runtime machine's OS is x64 (IMAGE_FILE_MACHINE_AMD64).
  - *arm64*: This package is used when the runtime machine's OS is ARM64 (IMAGE_FILE_MACHINE_ARM64).
  </BlockReference>

  <BlockReference
    content={{
      title: "RepairCondition",
      type: "string"
    }}
  >
  A condition that determines during Repair if the package will be repaired by default.
   This condition can use built-in variables and variables returned by searches.
  </BlockReference>

  <BlockReference
    content={{
      title: "RollbackLogPathVariable",
      type: "string"
    }}
  >
  Name of a Variable that will hold the path to the log file used during rollback. An empty value will cause
   the variable to not be set. The default is "WixBundleRollbackLog_[PackageId]" except for MSU packages which
   default to no logging.
  </BlockReference>

  <BlockReference
    content={{
      title: "Slipstream",
      type: "yesno"
    }}
  >
  Specifies whether to automatically slipstream the patch for any target msi packages in the chain. The default is "no".
   Even when the value is "no", you can still author the SlipstreamMsp element under MsiPackage elements as desired.
  </BlockReference>

  <BlockReference
    content={{
      title: "SourceFile",
      type: "string"
    }}
  >
  Location of the package to add to the bundle.
   The default value is the Name attribute, if provided.
   At a minimum, the SourceFile or Name attribute must be specified.
   Must not be specified if the package payload is given by the *PackagePayload element.
  </BlockReference>

  <BlockReference
    content={{
      title: "Vital",
      type: "yesno"
    }}
  >
  Specifies whether the package must succeed for the chain to continue. The default "yes"
   indicates that if the package fails then the chain will fail and roll back or stop. If
   "no" is specified then the chain will continue even if the package reports failure.
  </BlockReference>

</BlockWrapper>
