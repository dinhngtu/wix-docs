Metadata required to add a dependency on a package.


## Attributes

<BlockWrapper>

  <BlockReference
    content={{
      title: "Id",
      type: "string",
      required: true
    }}
  >
  Identity of the MSIX package to depend upon.
  </BlockReference>

  <BlockReference
    content={{
      title: "MaximumMajorVersionTested",
      type: "integer"
    }}
  >
  An integer indicating the highest major version of the dependency tested with this package.
  </BlockReference>

  <BlockReference
    content={{
      title: "Optional",
      type: "wxs:YesNoTypeUnion"
    }}
  >
  Indicates whether the dependency is optional (yes) or required (no).
  </BlockReference>

  <BlockReference
    content={{
      title: "Publisher",
      type: "string",
      required: true
    }}
  >
  Publisher of the MSIX package to depend upon.
  </BlockReference>

  <BlockReference
    content={{
      title: "Version",
      type: "string",
      required: true
    }}
  >
  Version of the MSIX package to depend upon.
  </BlockReference>

</BlockWrapper>
